    v_sfAngle.def("__repr__", [](sf::Angle& self) {
        std::stringstream ss;
        ss << "Angle(degree: " << self.asDegrees() << ", radian: " << self.asRadians() << ")";
        return ss.str();
    });
    m_sf.def("positiveRemainder", [](float a, float b) { return sf::priv::positiveRemainder(a, b); }, py::arg("a"), py::arg("b"));
    v_sfAngle.attr("Zero") = sf::Angle::Zero;
